# Clay waterfall: Job postings → enrichment → outreach trigger def monitor_signals(): # 1. Scrape LinkedIn jobs for CC manager postings jobs = linkedin.search_jobs(title="contact center manager", posted_within_days=7) for job in jobs: company = apollo.enrich_company(job.company_domain) if company.agent_count > 100: contact = apollo.find_contact(company.domain, title="VP Operations") instantly.add_to_sequence(contact, seq="hiring_signal_v1") log_to_airtable(company, contact, signal="hiring") # Runs daily via GitHub Actions cron
