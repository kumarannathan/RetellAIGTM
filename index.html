from flask import Flask, request import apollo, instantly, retell @app.route('/webhook/rb2b', methods=['POST']) def handle_visitor(): data = request.json domain = data['company']['domain'] visits = get_visit_count(domain, page='/pricing') if visits >= 3: contact = apollo.find_contact(domain, title='VP Operations') instantly.add_to_sequence(contact, seq='pricing_intent_v1') retell.schedule_call(contact.phone, delay_hours=48) return {'status': 'triggered', 'contact': contact.email} return {'status': 'watching', 'visits': visits}
